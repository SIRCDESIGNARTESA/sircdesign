Linha em branco def footer(self): # Rodapé de cada página (opcional) self.set_y(-15) self.set_font('Arial', 'I', 8) self.cell(0, 10, f'Página {self.page_no()}', 0, 0, 'C') def add_pedido_block(self, pedido_num, nome_cliente, telefone, endereco, estabelecimento, tipo): #
